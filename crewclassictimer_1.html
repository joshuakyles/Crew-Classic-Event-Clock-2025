        function updateTime() {

            const now = new Date();

            document.getElementById('time').textContent = now.toLocaleTimeString('en-GB', { hour12: false });

            document.getElementById('date').textContent = now.toDateString();

        }

        setInterval(updateTime, 1000);

        updateTime();


        const events = [

            { number: 1, date: 'Friday, March 28, 2025', time: '2:30 PM', name: 'Club Mx Masters 8 - Heat 1' },

            { number: 2, date: 'Friday, March 28, 2025', time: '2:38 PM', name: 'Club Mx Masters 8 - Heat 2' },

            { number: 4, date: 'Friday, March 28, 2025', time: '2:54 PM', name: 'Club Novice Masters Mx8 - Final' },

            { number: 5, date: 'Friday, March 28, 2025', time: '3:02 PM', name: 'Womens Masters D Cup (50-54) - Final' },

            { number: 6, date: 'Friday, March 28, 2025', time: '3:10 PM', name: 'Mens Masters D (50-54) - Final' },

            { number: 7, date: 'Friday, March 28, 2025', time: '3:18 PM', name: 'Club Masters M4X - Heat 1' },

            { number: 8, date: 'Friday, March 28, 2025', time: '3:26 PM', name: 'Club Masters M4X - Heat 2' }

        ];

        

        for (let i = 9; i <= 200; i++) {

            events.push({ number: i, date: 'TBD', time: 'TBD', name: `Event ${i}` });

        }


        let currentIndex = 0;

        

        function prevEvent() {

            if (currentIndex > 0) {

                currentIndex--;

                displayEvent();

            }

        }

        

        function nextEvent() {

            if (currentIndex < events.length - 1) {

                currentIndex++;

                displayEvent();

            }

        }

        

        const eventSelect = document.getElementById('eventSelect');

        events.forEach((event, index) => {

            let option = document.createElement('option');

            option.value = index;

            option.textContent = `#${event.number} - ${event.date} - ${event.time} - ${event.name}`;

            eventSelect.appendChild(option);

        });

        

        function selectEvent() {

            const selectedIndex = eventSelect.value;

            if (selectedIndex !== "") {

                currentIndex = parseInt(selectedIndex);

                displayEvent();

            }

        }

        

        function displayEvent() {

            document.getElementById('eventDetails').innerHTML = `<p><strong>#${events[currentIndex].number} - ${events[currentIndex].date}</strong><br>${events[currentIndex].time} - ${events[currentIndex].name}</p>`;

            eventSelect.value = currentIndex;

        }

        

        function autoUpdateEvent() {

            const now = new Date();

            const currentTime = now.toLocaleTimeString('en-US', { hour: '2-digit', minute: '2-digit', hour12: true });

            const currentDate = now.toDateString();

            for (let i = 0; i < events.length; i++) {

                if (events[i].date === currentDate && events[i].time === currentTime) {

                    currentIndex = i;

                    displayEvent();

                    break;

                }

            }

        }

        setInterval(autoUpdateEvent, 60000);

        displayEvent();
